function linreg(pts){
  var n=pts.length,sx=0,sy=0,sxx=0,sxy=0;
  for(var i=0;i<n;i++){sx+=pts[i].x;sy+=pts[i].y;sxx+=pts[i].x*pts[i].x;sxy+=pts[i].x*pts[i].y;}
  var m=(n*sxy-sx*sy)/(n*sxx-sx*sx),b=(sy-m*sx)/n;
  var ybar=sy/n,sst=0,ssr=0;
  for(var i=0;i<n;i++){sst+=Math.pow(pts[i].y-ybar,2);ssr+=Math.pow(pts[i].y-(m*pts[i].x+b),2);}
  return {m:m,b:b,r2:1-ssr/sst};
}
var axCol='#555', gridCol='rgba(0,0,0,0.06)';
function scXY(xt,yt){
  return {
    x:{title:{display:true,text:xt,color:axCol,font:{size:11,family:'DM Mono'}},
       grid:{color:gridCol},ticks:{color:axCol,font:{size:10}},border:{color:'#333',width:1}},
    y:{title:{display:true,text:yt,color:axCol,font:{size:11,family:'DM Mono'}},
       grid:{color:gridCol},ticks:{color:axCol,font:{size:10}},border:{color:'#333',width:1}}
  };
}
